## 三、config.h 模板

```c
#ifndef __CONFIG_H
#define __CONFIG_H
#include <stdint.h>

/* ============ 1. 系统时钟 ============ */
#define SYS_CLK_HZ              80000000UL    /* MSPM0G3507 max 80MHz */
#define SYSTICK_HZ              1000U         /* 1 ms */

/* ============ 2. 电机 / 编码器 ============ */
#define WHEEL_DIAMETER_MM       65.0f
#define WHEEL_BASE_MM           155.0f
#define ENCODER_PPR             1320          /* 11×4×30 */
#define MM_PER_PULSE            (3.14159265f * WHEEL_DIAMETER_MM / ENCODER_PPR)

/* PWM */
#define PWM_FREQ_HZ             10000U        /* TB6612 100kHz 上限 */
#define PWM_PERIOD              ((SYS_CLK_HZ / PWM_FREQ_HZ) - 1)
#define PWM_DUTY_MAX            ((PWM_PERIOD) - 1)

/* ============ 3. PID ============ */
#define PID_SPEED_KP            10.0f
#define PID_SPEED_KI            1.0f
#define PID_SPEED_KD            0.0f
#define PID_YAW_KP              3.0f
#define PID_YAW_KI              0.05f
#define PID_YAW_KD              1.5f
#define PID_TURN_KP             4.0f
#define PID_TURN_KI             0.0f
#define PID_TURN_KD             2.0f

#define PID_YAW_OUT_MIN         -300.0f
#define PID_YAW_OUT_MAX          300.0f

/* ============ 4. 速度 ============ */
#define V_LINE_PWM              500
#define V_TURN_PWM              300
#define V_DECEL_DIST_MM         150.0f
#define V_LINE_END_PWM          250

/* ============ 5. 任务调度 ============ */
#define CTRL_SPEED_HZ           100   /* 速度环 10ms */
#define CTRL_TRACK_HZ           50    /* 循迹 20ms */
#define DISP_PERIOD_MS          200

/* ============ 6. 路径 ============ */
#define PATH_LINE_AB_MM         1000.0f
#define PATH_ARC_MM             1256.6f
#define PATH_TURN_AC_DEG        38.66f

#endif
```

## 四、main.c 状态机骨架

```c
#include <ti/driverlib/driverlib.h>
#include "ti_msp_dl_config.h"   /* SysConfig 自动生成 */
#include "config.h"

typedef enum {
    ST_IDLE = 0,
    ST_PRECHECK,
    ST_RUNNING,
    ST_PAUSE,
    ST_DONE,
    ST_FAULT,
} CarState;

static volatile CarState g_state = ST_IDLE;
static volatile uint32_t systick_ms = 0;

void SysTick_Handler(void)   /* 这里用 driverlib 的 SysTick API */
{
    systick_ms++;
}

int main(void)
{
    SYSCFG_DL_init();   /* SysConfig 一次配齐：CLK/GPIO/TIM/ADC/I2C/UART */
    SysTick_Config(SYS_CLK_HZ / SYSTICK_HZ);

    Motor_Init();
    Encoder_Init();
    IRTrack_Init();
    IMU_Init();
    Display_Init();

    /* 调用 IMU 静态平均 200 次得零偏 */
    IMU_Calibrate(200);

    Path_Load(/*task_id=*/2);

    uint32_t last_speed_ms = 0;
    uint32_t last_track_ms = 0;
    uint32_t last_disp_ms  = 0;

    while (1) {
        uint32_t now = systick_ms;

        /* 10 ms 速度环 */
        if (now - last_speed_ms >= 10) {
            last_speed_ms = now;
            int32_t pulse_l = Encoder_GetCount(ENC_LEFT);
            int32_t pulse_r = Encoder_GetCount(ENC_RIGHT);
            float v_l = pulse_l * MM_PER_PULSE * 100.0f;   /* mm/s */
            float v_r = pulse_r * MM_PER_PULSE * 100.0f;
            int16_t pwm_l = (int16_t)PID_SpeedUpdate(&pid_v_l, v_l);
            int16_t pwm_r = (int16_t)PID_SpeedUpdate(&pid_v_r, v_r);
            Motor_SetPWM(MOTOR_LEFT,  pwm_l);
            Motor_SetPWM(MOTOR_RIGHT, pwm_r);
        }

        /* 20 ms 循迹 / 航向 */
        if (now - last_track_ms >= 20) {
            last_track_ms = now;
            float yaw = IMU_ReadYaw();
            int ir[5];
            IRTrack_Read(ir);
            float pos = Track_CalcPosition(ir);
            /* 直线段 用 yaw_pid，弧线段 用 track_pid */
            float diff = (g_seg_type == SEG_LINE) ?
                         PID_YawUpdate(&pid_yaw, yaw) :
                         PID_TrackUpdate(&pid_track, pos);
            PID_SetSpeedTarget(&pid_v_l, V_LINE_PWM - diff);
            PID_SetSpeedTarget(&pid_v_r, V_LINE_PWM + diff);

            /* 段切换 */
            if (Path_SegmentDone()) Path_Next();
            if (Path_AllDone()) g_state = ST_DONE;
        }

        /* 200 ms 显示 */
        if (now - last_disp_ms >= 200) {
            last_disp_ms = now;
            Display_Update(g_state, /* meas */);
        }
    }
}
```

## 五、SysConfig（.syscfg）必勾清单

| 模块 | 配置 |
|---|---|
| **SYSCTL** | MCLK = 80MHz (HFCLK + 4×PLL); ULPCLK = 40MHz |
| **TIMA0** | PWM Mode, 10kHz, 4 通道（左 PWM/IN1/IN2, 右 PWM/IN1/IN2）|
| **TIMG6/TIMG7** | Encoder Mode（QEI），计数 A/B 通道 |
| **ADC0** | 12bit, 2MSPS, Hardware trigger=TIMA0 CC0, 多通道扫描 |
| **I2C0** | 400kHz, Master |
| **UART0** | 115200 8N1, FIFO 8 byte |
| **GPIO** | 5 路 IR 数字输入下拉, 4 个按键上拉, LED 推挽 |
| **SysTick** | 80MHz / 1000 = 80000 reload |

## 八、5 大踩坑

1. **SysConfig 没生成 ti_msp_dl_config** → 重新打开 .syscfg 双击 Save
2. **TB6612 STBY 未拉高** → 默认引脚浮空，PWM 不输出；SysConfig 里设 GPIO Output High
3. **编码器方向反** → SysConfig 里 TIMG QEI 的 Phase A/B 互换或加 `cnt = -cnt;`
4. **I2C 卡死** → 没接上拉电阻 4.7kΩ；MSPM0 内部上拉 IIC 不可靠
5. **printf 没输出** → CCS 默认 stdout 不指向 UART；需重定向 `_write` 函数

## 十、降级清单

| 降级 | 操作 | 影响 |
|---|---|---|
| 没有 IMU | 用纯编码器航向估计：`yaw = (pulse_r - pulse_l) × MM_PER_PULSE / WHEEL_BASE_MM` | 累积误差大，一圈内可用 |
| TIMG QEI 配置不通 | 改用纯 GPIO 中断计数 | CPU 负载升 30% |
| 5 路 IR 模拟版 | 单独引到 ADC0 5 通道 | 多 1 个调理电阻 + ADC |

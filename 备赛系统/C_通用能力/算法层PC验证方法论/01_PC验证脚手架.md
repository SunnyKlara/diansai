## 二、cal_helper.py 完整骨架

```python
"""
cal_helper.py
=============

PC 端校准 / 调试助手（与 MCU 通过 UART 对话）。

工作流：
  1. python cal_helper.py --port COM3 --stat       # 看实时数据
  2. python cal_helper.py --port COM3 --calibrate  # 交互式校准
  3. python cal_helper.py --port COM3 --dump x.csv # 导出原始数据
  4. python cal_helper.py --port COM3 --verify x.csv # 跑金标准对比
"""

import argparse
import sys
import time
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
import algo_reference as alg


def open_serial(port: str, baud: int = 115200):
    try:
        import serial
    except ImportError:
        print("ERR: 缺少 pyserial，请 `pip install pyserial`", file=sys.stderr)
        sys.exit(2)
    return serial.Serial(port, baudrate=baud, timeout=2)


def cmd(ser, line: str, wait_s: float = 0.5) -> str:
    ser.reset_input_buffer()
    ser.write((line + "\r\n").encode())
    ser.flush()
    time.sleep(wait_s)
    return ser.read_all().decode(errors="replace")


def do_stat(ser):
    print(cmd(ser, "STAT"))


def do_calibrate(ser):
    print("[1] 当前实测：")
    print("    " + cmd(ser, "STAT").strip())
    real_val = float(input("[2] 输入标准基准值: "))
    resp = cmd(ser, f"CAL VALUE={real_val}")
    print("    " + resp.strip())


def do_dump(ser, out_csv: str, n_samples: int):
    ser.write(b"DUMP\r\n")
    ser.flush()
    lines = []
    started = False
    deadline = time.time() + 30.0
    while time.time() < deadline:
        line = ser.readline().decode(errors="replace").strip()
        if not line:
            continue
        if line == "DUMP_START":
            started = True; continue
        if line == "DUMP_END":
            break
        if started:
            lines.append(line)
    
    if len(lines) < n_samples:
        print(f"ERR: 期望 {n_samples} 行，实际 {len(lines)} 行", file=sys.stderr)
        sys.exit(3)
    
    header = "raw_value\n"     # 按题目调整
    Path(out_csv).write_text(header + "\n".join(lines[:n_samples]) + "\n",
                             encoding="utf-8")
    print(f"OK 写入 {out_csv}")


def do_verify(csv_path: str):
    """用 algo_reference 跑金标准 → 输出对比"""
    rows = Path(csv_path).read_text(encoding="utf-8").strip().splitlines()[1:]
    raw = [int(r) for r in rows]
    # ... 调用 alg.xxx_function(raw) 输出对比表


def do_reset(ser):
    print(cmd(ser, "RST"))


def main():
    ap = argparse.ArgumentParser(description="<题号 + 题名> PC 校准助手")
    ap.add_argument("--port", help="串口号")
    ap.add_argument("--baud", type=int, default=115200)
    g = ap.add_mutually_exclusive_group(required=True)
    g.add_argument("--stat", action="store_true")
    g.add_argument("--calibrate", action="store_true")
    g.add_argument("--dump", metavar="CSV")
    g.add_argument("--verify", metavar="CSV")
    g.add_argument("--reset", action="store_true")
    args = ap.parse_args()

    if args.verify:
        do_verify(args.verify); return

    if not args.port:
        print("ERR: 该子命令需要 --port", file=sys.stderr); sys.exit(2)
    
    ser = open_serial(args.port, args.baud)
    try:
        if args.stat: do_stat(ser)
        elif args.calibrate: do_calibrate(ser)
        elif args.dump: do_dump(ser, args.dump, alg.DFT_N)
        elif args.reset: do_reset(ser)
    finally:
        ser.close()


if __name__ == "__main__":
    main()
```

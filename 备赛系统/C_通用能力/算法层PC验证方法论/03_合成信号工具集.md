## 一、电气类信号

### 纯阻负载（电流 / 电压同相）

```python
def synth_pure_resistive(v_rms, r_load, freq, fs, n):
    """v(t) = √2·V·sin(ωt), i(t) = v(t)/R"""
    v, i = [], []
    v_peak = v_rms * math.sqrt(2)
    for k in range(n):
        t = k / fs
        vk = v_peak * math.sin(2 * math.pi * freq * t)
        v.append(vk)
        i.append(vk / r_load)
    return v, i
```

### 阻感负载（电流相位滞后）

```python
def synth_rl_load(v_rms, r, l_h, freq, fs, n):
    """z = R + jωL，电流相对电压滞后 atan(ωL/R)"""
    omega = 2 * math.pi * freq
    z_mag = math.sqrt(r**2 + (omega*l_h)**2)
    phi = math.atan2(omega*l_h, r)
    
    v_peak = v_rms * math.sqrt(2)
    i_peak = v_peak / z_mag
    
    v, i = [], []
    for k in range(n):
        t = k / fs
        v.append(v_peak * math.sin(omega * t))
        i.append(i_peak * math.sin(omega * t - phi))
    return v, i
```

### 非线性负载（基波 + 奇次谐波，模拟整流电源）

```python
def synth_nonlinear(v_rms, harmonic_amps, freq, fs, n):
    """harmonic_amps = {1: 1.0, 3: 0.3, 5: 0.15, 7: 0.07, 9: 0.03}
       (RMS 幅度，会自动转峰值)"""
    omega = 2 * math.pi * freq
    v_peak = v_rms * math.sqrt(2)
    v, i = [], []
    for k in range(n):
        t = k / fs
        v.append(v_peak * math.sin(omega * t))
        i_t = sum(a * math.sqrt(2) * math.sin(h * omega * t)
                  for h, a in harmonic_amps.items())
        i.append(i_t)
    return v, i
```

### 已知 THD 的合成信号（用于 THD 测量验证）

```python
def synth_thd_signal(freq, vpp, thd_percent, fs, n):
    """合成精确 THD 的信号：H3 占主，H5 次。
       H3 = a·H1, H5 = (a/3)·H1
       √(a² + (a/3)²) = thd_target
    """
    target = thd_percent / 100
    a = target / math.sqrt(1 + 1/9)
    a3, a5 = a, a/3
    
    v_peak = vpp / 2
    samples = []
    for k in range(n):
        t = k / fs
        v = (v_peak * math.sin(2*math.pi*freq*t)
             + a3*v_peak * math.sin(2*math.pi*3*freq*t)
             + a5*v_peak * math.sin(2*math.pi*5*freq*t))
        samples.append(v)
    return samples
```

## 二、控制类信号

### 直线运动学（差速车 PID 仿真用）

```python
@dataclass
class CarState:
    x: float = 0.0
    y: float = 0.0
    yaw_deg: float = 0.0
    v_left: float = 0.0
    v_right: float = 0.0

def step_kinematics(state: CarState, dt_s: float, wheel_base_mm: float):
    v = 0.5 * (state.v_left + state.v_right)
    omega_rad = (state.v_right - state.v_left) / wheel_base_mm
    yaw_rad = math.radians(state.yaw_deg)
    state.x += v * math.cos(yaw_rad) * dt_s
    state.y += v * math.sin(yaw_rad) * dt_s
    state.yaw_deg += math.degrees(omega_rad * dt_s)
    while state.yaw_deg > 180: state.yaw_deg -= 360
    while state.yaw_deg < -180: state.yaw_deg += 360
```

## 三、信号处理类

### 已知 H(s) 的滤波器（用于扫频测量验证）

```python
# 二阶低通
def lp_2nd_order(f, fc=5000, q=0.707):
    s = 1j * 2 * math.pi * f
    wc = 2 * math.pi * fc
    return 1.0 / (1 + s/(q*wc) + (s/wc)**2)

# 二阶高通
def hp_2nd_order(f, fc=1000, q=0.707):
    s = 1j * 2 * math.pi * f
    wc = 2 * math.pi * fc
    return (s/wc)**2 / (1 + s/(q*wc) + (s/wc)**2)

# 二阶带通
def bp_2nd_order(f, fc=2000, q=5.0):
    s = 1j * 2 * math.pi * f
    wc = 2 * math.pi * fc
    return (s/(q*wc)) / (1 + s/(q*wc) + (s/wc)**2)
```

### 加性高斯白噪声（用于鲁棒性测试）

```python
import random

def add_gaussian_noise(signal: List[float], snr_db: float) -> List[float]:
    """向信号加入指定 SNR 的高斯噪声。"""
    p_signal = sum(s*s for s in signal) / len(signal)
    p_noise = p_signal / (10 ** (snr_db / 10))
    sigma = math.sqrt(p_noise)
    return [s + random.gauss(0, sigma) for s in signal]
```

### ADC 量化（合成信号模拟硬件采样）

```python
def adc_quantize(signal: List[float], v_ref: float = 3.3,
                 bits: int = 12, offset_lsb: int = 2048) -> List[int]:
    """物理量 → ADC 原始码。"""
    full_scale = (1 << bits) - 1
    half = (1 << bits) // 2
    quantized = []
    for v in signal:
        # 假设输入范围 ±v_ref/2，映射到 0..full_scale
        adc = offset_lsb + int(round(v * half / (v_ref / 2)))
        adc = max(0, min(full_scale, adc))
        quantized.append(adc)
    return quantized
```

## 四、误差工具

```python
def err_pct(actual: float, expected: float) -> float:
    if abs(expected) < 1e-9:
        return 0.0
    return (actual - expected) / expected * 100.0


def err_abs(actual: float, expected: float) -> float:
    return actual - expected


def rms_error(actual: List[float], expected: List[float]) -> float:
    """两序列的 RMSE。"""
    n = min(len(actual), len(expected))
    return math.sqrt(sum((actual[k] - expected[k])**2 for k in range(n)) / n)
```

## 五、断言模板

```python
def assert_within(actual: float, expected: float, tol_pct: float, name: str):
    err = err_pct(actual, expected)
    if abs(err) > tol_pct:
        raise AssertionError(
            f"{name}: actual={actual:.4f}, expected={expected:.4f}, "
            f"err={err:+.4f}% > tol={tol_pct}%")
    print(f"  ✓ {name}: err={err:+.4f}% < {tol_pct}%")


# 使用示例
```

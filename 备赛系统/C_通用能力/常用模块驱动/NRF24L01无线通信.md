# NRF24L01无线通信 —— 双车协同必备

## 一、模块参数
- 频率：2.4GHz ISM频段
- 速率：250kbps / 1Mbps / 2Mbps
- 距离：普通版约50m，PA版约1000m
- 接口：SPI
- 供电：3.3V（5V会烧！）

## 二、接线

| NRF24L01 | STM32 | 说明 |
|----------|-------|------|
| VCC | 3.3V | **不能接5V！** |
| GND | GND | |
| CE | PB0 | 使能（GPIO输出） |
| CSN | PB1 | 片选（GPIO输出） |
| SCK | PA5 | SPI1_SCK |
| MOSI | PA7 | SPI1_MOSI |
| MISO | PA6 | SPI1_MISO |
| IRQ | PA4 | 中断（可选） |

## 三、驱动代码框架

```c
#include "nrf24l01.h"

// SPI读写一个字节
static uint8_t SPI_RW(uint8_t byte) {
    uint8_t rx;
    HAL_SPI_TransmitReceive(&hspi1, &byte, &rx, 1, 10);
    return rx;
}

// 写寄存器
void NRF_WriteReg(uint8_t reg, uint8_t val) {
    CSN_LOW();
    SPI_RW(0x20 | reg);  // 写命令
    SPI_RW(val);
    CSN_HIGH();
}

// 读寄存器
uint8_t NRF_ReadReg(uint8_t reg) {
    CSN_LOW();
    SPI_RW(reg);  // 读命令
    uint8_t val = SPI_RW(0xFF);
    CSN_HIGH();
    return val;
}

// 初始化为发送模式
void NRF_InitTX(void) {
    CE_LOW();
    NRF_WriteReg(CONFIG, 0x0E);      // 发送模式，使能CRC
    NRF_WriteReg(EN_AA, 0x01);       // 通道0自动应答
    NRF_WriteReg(EN_RXADDR, 0x01);   // 使能通道0
    NRF_WriteReg(SETUP_AW, 0x03);    // 5字节地址
    NRF_WriteReg(SETUP_RETR, 0x1A);  // 500μs重发，10次
    NRF_WriteReg(RF_CH, 40);         // 频道40
    NRF_WriteReg(RF_SETUP, 0x07);    // 1Mbps，0dBm
    // 设置发送地址和接收地址（5字节）
    uint8_t addr[5] = {0x34, 0x43, 0x10, 0x10, 0x01};
    NRF_WriteBuf(TX_ADDR, addr, 5);
    NRF_WriteBuf(RX_ADDR_P0, addr, 5);
}

// 发送数据（最多32字节）
uint8_t NRF_SendData(uint8_t* data, uint8_t len) {
    CE_LOW();
    CSN_LOW();
    SPI_RW(0xA0);  // 写TX FIFO
    for (int i = 0; i < len; i++) SPI_RW(data[i]);
    CSN_HIGH();
    CE_HIGH();
    HAL_Delay(1);  // 等待发送
    CE_LOW();
    
    uint8_t status = NRF_ReadReg(STATUS);
    NRF_WriteReg(STATUS, 0x70);  // 清除标志
    
    if (status & 0x20) return 0;  // 发送成功
    return 1;  // 发送失败
}

// 初始化为接收模式
void NRF_InitRX(void) {
    CE_LOW();
    NRF_WriteReg(CONFIG, 0x0F);      // 接收模式
    NRF_WriteReg(RX_PW_P0, 32);      // 接收数据宽度32字节
    // 其他配置同TX
    CE_HIGH();  // 开始接收
}

// 接收数据
uint8_t NRF_RecvData(uint8_t* data) {
    uint8_t status = NRF_ReadReg(STATUS);
    if (status & 0x40) {  // 收到数据
        CSN_LOW();
        SPI_RW(0x61);  // 读RX FIFO
        for (int i = 0; i < 32; i++) data[i] = SPI_RW(0xFF);
        CSN_HIGH();
        NRF_WriteReg(STATUS, 0x70);
        return 0;  // 成功
    }
    return 1;  // 无数据
}
```

## 四、双车通信示例

```c
// 领头车（发送端）
typedef struct {
    uint8_t header;     // 0xAA
    uint8_t cmd;        // 命令
    int16_t speed;      // 速度
    int16_t distance;   // 距离
    uint8_t checksum;
} Packet_t;

void leader_send(void) {
    Packet_t pkt;
    pkt.header = 0xAA;
    pkt.cmd = CMD_RUN;
    pkt.speed = current_speed;
    pkt.distance = encoder_count;
    pkt.checksum = pkt.header ^ pkt.cmd ^ (pkt.speed&0xFF) ^ (pkt.distance&0xFF);
    
    NRF_SendData((uint8_t*)&pkt, sizeof(pkt));
}

// 跟随车（接收端）
void follower_recv(void) {
    uint8_t buf[32];
    if (NRF_RecvData(buf) == 0) {
        Packet_t* pkt = (Packet_t*)buf;
        if (pkt->header == 0xAA) {
            // 校验
            uint8_t chk = pkt->header ^ pkt->cmd ^ (pkt->speed&0xFF) ^ (pkt->distance&0xFF);
            if (chk == pkt->checksum) {
                process_command(pkt);
            }
        }
    }
}
```

## 五、常见问题

| 问题 | 原因 | 解决 |
|------|------|------|
| 完全不通信 | VCC接了5V | 换3.3V供电 |
| 通信不稳定 | 电源纹波大 | VCC加100μF电容 |
| 距离很短 | 天线被遮挡 | 天线朝上，远离金属 |
| 丢包严重 | 频道干扰 | 换一个RF_CH值 |
